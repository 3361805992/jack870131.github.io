Reverse Linked List

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/*
 * Recursive solution
 * 1. find the new head at the tail of the linked list
 * 2. the pointer of the next node point to the previous node
 * 3. the pointer of the current node point to null to cut the connection with the next node
 */
class Solution {
  public ListNode reverseList(ListNode head) {
    if (head == null || head.next == null) return head;
    ListNode newHead = reverseList(head.next);
    head.next.next = head;
    head.next = null;
    return newHead;
  }
}
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/*
 * Iterative solution
 * 1. the temp pointer records the position of the next node
 * 2. the pointer of the current node point to the previous node
 * 3. the pointer of the previous node shift to the next node
 * 4. the pointer of the current node move to the position of the temp pointer 
 */
class Solution {
  public ListNode reverseList(ListNode head) {
    ListNode cur = head;
    ListNode prev = null;
    while (cur != null) {
      ListNode tmp = cur.next;
      cur.next = prev;
      prev = cur;
      cur = tmp;
    }
    return prev;
  }
}
