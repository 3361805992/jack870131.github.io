Valid Parentheses

    class Solution {
      /*
       * '(', ')', '{', '}', '[', ']'
       * 1. init a stack for storing open brackets
       * 2. init a HashMap<closeBracket, openBracket>
       * 3. if equal the open bracket, push it into the stack
       * 4. if equal the close bracket, pop the stack and see if it matches the value
       */
      public boolean isValid(String s) {
        if (s == null) {
          return false;
        }
        HashMap<Character, Character> map = new HashMap<>();
        map.put('(', ')');
        map.put('[', ']');
        map.put('{', '}');
        Stack<Character> stack = new Stack<>();
        for (int i = 0; i < s.length(); i++) {
          char character = s.charAt(i);
          if (character == ')' || character == ']' || character == '}') {
            if (stack.empty() || character != map.get(stack.pop())) {
              return false;
            }
          }
          if (character == '(' || character == '[' || character == '{') {
            stack.push(character);
          }
        }
        return stack.isEmpty();
      }
    }
class Solution {
  public boolean isValid(String s) {
    Stack<Character> stack = new Stack<Character>();
    for (char c : s.toCharArray()) {
      if (c == '(')
        stack.push(')');
      else if (c == '{')
        stack.push('}');
      else if (c == '[')
        stack.push(']');
      else if (stack.isEmpty() || stack.pop() != c)
        return false;
    }
    return stack.isEmpty();
  }
}
