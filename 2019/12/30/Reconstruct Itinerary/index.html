Reconstruct Itinerary

class Solution {
  public List<String> findItinerary(List<List<String>> tickets) {
    // build graph
    Map<String, LinkedList<String>> map = new HashMap<>();
    tickets.forEach(
            ticket -> map
                    .computeIfAbsent(ticket.get(0), toList -> new LinkedList<>())
                    .add(ticket.get(1))
    );
    map.forEach((k, v) -> Collections.sort(v));
    LinkedList<String> route = new LinkedList<>();
    dfs(map, route, "JFK");
    return route;
  }

  private boolean dfs(Map<String, LinkedList<String>> map, LinkedList<String> route, String source) {
    if (map.isEmpty()) // all sources exhausted
      return route.add(source);
    if (!map.containsKey(source)) // not a valid source
      return false;
    LinkedList<String> destinations = map.get(source);
    for (int i = 0; i < destinations.size(); i++) {
      String to = destinations.get(i);

      route.addLast(source);
      destinations.remove(i);
      if (map.get(source).isEmpty())
        map.remove(source);

      if (dfs(map, route, to))
        return true;
      // revert the last 3 steps done before the DFS i.e. restore the map and push back the to-node
      map.computeIfAbsent(source, toList -> new LinkedList<>()).add(i, to);
      route.removeLast();
    }
    return false;
  }
}
class Solution {
  public List<String> findItinerary(List<List<String>> tickets) {
    Map<String, PriorityQueue<String>> map = new HashMap<>();
    tickets.forEach(
            ticket -> map
                  .computeIfAbsent(ticket.get(0), q -> new PriorityQueue<>())
                  .add(ticket.get(1))
    );
    LinkedList<String> route = new LinkedList<>();
    dfs(map, route, "JFK");
    return route;
  }

  private void dfs(Map<String, PriorityQueue<String>> map, LinkedList<String> route, String source) {
    while (map.containsKey(source) && !map.get(source).isEmpty())
      dfs(map, route, map.get(source).poll());
    route.addFirst(source);
  }
}
